\begin{frame}{¿Por qué la distancia euclidiana basta?}
\small
\begin{columns}[T]

\begin{column}{0.33\textwidth}
\setbeamercolor{block title}{fg=white,bg=uaiblue}
\begin{block}{Hipótesis del manifold}
Los datos viven cerca de una variedad de baja dimensión.
\vspace{0.3em}\\
{\scriptsize Inliers $\approx$ buenas muestras.}
\end{block}
\end{column}

\begin{column}{0.33\textwidth}
\setbeamercolor{block title}{fg=white,bg=uaiblue}
\begin{block}{Concentración de la medida}
En $\mathbb{R}^{768}$ la similitud coseno se concentra en un rango muy estrecho.
\vspace{0.3em}\\
{\scriptsize \cita{Tessari et al.\ 2024}}
\end{block}
\end{column}

\begin{column}{0.33\textwidth}
\setbeamercolor{block title}{fg=white,bg=uaiblue}

\end{block}
\end{column}

\end{columns}
